#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <SPI.h>
#include <Keypad.h>
#include "TFT_KIX/TFT_eSPI.h"

/* --- KEYPAD CONFIGURATION (Safe Pins for T-Display V1.1) --- */
const byte ROWS = 4; 
const byte COLS = 4; 
char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
// Rows on Right side (33, 25, 26, 27), Cols on Left side (15, 13, 12, 2)
byte rowPins[ROWS] = {33, 25, 26, 27}; 
byte colPins[COLS] = {15, 13, 12, 2}; 
Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

/* --- CONFIGURATION --- */
const char* ssid = "TP-Link_46C2";
const char* password = "80165897";
const char* coinapi_key = "3e97fe7f-44be-4fec-939c-e59a9342703b";
String lnbits_url = "http://34.226.196.147:5004/api/v1/payments";
String lnbits_api_key = "2b273ca3d3a84db4a3857ff54776aa96";

/* --- GLOBAL STATE --- */
String current_payment_hash = "";
String input_amount = "";
TFT_eSPI tft = TFT_eSPI();

/* --- HARDWARE & DISPLAY HELPERS --- */
void kickstartHardware() {
  pinMode(4, OUTPUT); digitalWrite(4, HIGH); // Backlight
  pinMode(23, OUTPUT); digitalWrite(23, HIGH); // Power
}

void logDisplay(String msg, uint16_t color = 0xFFFF) {
  tft.fillRect(0, 110, 240, 25, 0x0000); // Clear log line
  tft.setCursor(0, 115);
  tft.setTextColor(color, 0x0000);
  tft.setTextSize(1);
  tft.println("> " + msg);
  Serial.println("[LOG] " + msg);
}

void displayInput() {
  tft.fillRect(0, 40, 240, 50, 0x0000);
  tft.setCursor(20, 55);
  tft.setTextColor(0xFFFF, 0x0000);
  tft.setTextSize(4);
  tft.print(input_amount + " SAT");
}

void connectWiFi() {
  tft.fillScreen(0x0000);
  tft.setTextColor(0x07E0, 0x0000);
  tft.setTextSize(2);
  tft.setCursor(0, 10);
  tft.println("PROJECT KIX v1.0");
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  logDisplay("WIFI ONLINE", 0x07E0);
  displayInput();
}

/* --- API LOGIC --- */
void createInvoice(int sats) {
  logDisplay("FETCHING BOLT11...", 0x07FF);
  HTTPClient http;
  http.begin(lnbits_url);
  http.addHeader("X-Api-Key", lnbits_api_key);
  http.addHeader("Content-Type", "application/json");
  String payload = "{\"out\": false, \"amount\":" + String(sats) + ", \"memo\": \"KIX-DEBUG\"}";
  
  int httpCode = http.POST(payload);
  if (httpCode == 200) {
    DynamicJsonDocument doc(4096);
    deserializeJson(doc, http.getStream());
    String bolt11 = doc["payment_request"].as<String>();
    current_payment_hash = doc["payment_hash"].as<String>();
    
    // Debugging output
    Serial.println("\n--- NEW INVOICE ---");
    Serial.println(bolt11);
    Serial.println("-------------------\n");
    
    tft.fillScreen(0x0000);
    tft.setCursor(0, 10); tft.setTextColor(0x07FF);
    tft.setTextSize(1);
    tft.println("INVOICE GENERATED!");
    tft.println("CHECK SERIAL MONITOR FOR BOLT11");
    tft.setCursor(0, 50); tft.setTextColor(0xFFFF);
    tft.println("Press 'C' to Verify Payment.");
  } else {
    logDisplay("ERR: " + String(httpCode), 0xF800);
  }
  http.end();
}

void manualCheck() {
  if (current_payment_hash == "") { logDisplay("NO ACTIVE HASH", 0xF800); return; }
  logDisplay("CHECKING...", 0xFFFF);
  HTTPClient http;
  http.begin(lnbits_url + "/" + current_payment_hash);
  http.addHeader("X-Api-Key", lnbits_api_key);
  if (http.GET() == 200) {
    DynamicJsonDocument doc(2048);
    deserializeJson(doc, http.getStream());
    if (doc["paid"].as<bool>()) {
      tft.fillScreen(0x07E0); tft.setTextColor(0x0000);
      tft.setCursor(30, 55); tft.setTextSize(3);
      tft.println("PAID!");
      delay(4000);
      connectWiFi();
    } else {
      logDisplay("NOT PAID YET", 0xFFE0);
    }
  }
  http.end();
}

void checkBalance() {
  logDisplay("GETTING WALLET...", 0xFFE0);
  HTTPClient http;
  http.begin("http://34.226.196.147:5004/api/v1/wallet");
  http.addHeader("X-Api-Key", lnbits_api_key);
  if (http.GET() == 200) {
    String raw = http.getString();
    tft.fillScreen(0x0000);
    tft.setCursor(0, 10); tft.setTextColor(0xFFE0); tft.println("BALANCE RAW:");
    tft.setCursor(0, 30); tft.setTextColor(0xFFFF); tft.println(raw.substring(0, 200));
    delay(5000);
    connectWiFi();
  }
  http.end();
}

/* --- MAIN SETUP & LOOP --- */
void setup() {
  Serial.begin(115200);
  kickstartHardware();
  tft.init(); 
  tft.setRotation(1); 
  tft.fillScreen(0x0000);
  connectWiFi();
}

void loop() {
  char key = keypad.getKey();
  
  if (key) {
    Serial.print("Key Detected: "); Serial.println(key);
    
    if (key >= '0' && key <= '9') {
      input_amount += key;
      displayInput();
    }
    else if (key == 'A') checkBalance();
    else if (key == 'C') manualCheck();
    else if (key == 'D') ESP.restart();
    else if (key == '*') {
      input_amount = "";
      displayInput();
      logDisplay("CLEARED", 0xFFE0);
    }
    else if (key == '#') {
      int amt = input_amount.toInt();
      if (amt > 0) {
        createInvoice(amt);
        input_amount = ""; // Reset buffer
      }
    }
  }
}
